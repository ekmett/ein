144constexpr auto cmpint(T a, T b) noexcept {
145 if constexpr (imm8 == CMPINT::TRUE) return -1;
146 else if constexpr (imm8 == CMPINT::FALSE) return 0;
147 else if constexpr (imm8 == CMPINT::LT) return a < b;
148 else if constexpr (imm8 == CMPINT::NLT) return a >= b;
149 else if constexpr (imm8 == CMPINT::LE) return a <= b;
