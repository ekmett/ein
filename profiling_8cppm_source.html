87struct profile_event {
88 using duration = Duration;
89 using clock = Clock;
90 static_c_string name{}; // null
91 static_c_string cat{}; // null
92 event_type ph;
93 size_t ts = std::chrono::duration_cast<duration>(clock::now().time_since_epoch()).count();
94 pid_t pid = getpid();
95 size_t tid = std::hash<std::thread::id>{}(std::this_thread::get_id());
96 scope s;
97 std::optional<string> id{}; // null // an actual real string
98 static_c_string bp{}; // null
99 json args{}; // empty object
100};

102export template <typename Duration, typename Clock>
103void to_json(json& j, const profile_event<Duration, Clock> & t) {
104 if (not t.name) j["name"] = t.name;
105 if (not t.cat) j["cat"] = t.cat;
106 j["ph"] = t.ph;
107 j["ts"] = t.ts;
108 j["pid"] = t.pid;
109 j["tid"] = t.tid;
110 if (t.s != scope::none) j["s"] = t.s;
111 if (t.id.has_value()) j["id"] = t.id;
112 if (t.bp) j["bp"] = t.bp;
113 if (not t.args.is_null()) j["args"] = t.args;
114}
