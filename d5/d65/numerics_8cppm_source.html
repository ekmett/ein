72constexpr T scalef(T x, T y) noexcept {
73 if consteval {
74 // Constexpr path using bit manipulation
75
76 using uint_type = uint_t<T>; // std::conditional_t<std::is_same_v<T, float>, uint32_t, uint64_t>;
77 constexpr int exponent_bias = std::is_same_v<T, float> ? 127 : 1023;
78 constexpr uint_type exponent_mask = std::is_same_v<T, float> ? 0x7F800000 : 0x7FF0000000000000ull;
79 constexpr uint_type mantissa_mask = std::is_same_v<T, float> ? 0x007FFFFF : 0x000FFFFFFFFFFFFFull;
80
81 if (x == 0.0 || isnan(x) || std::isinf(x))
82 return x; // Handle special cases
83
84 // Get the raw bits of the floating-point number
85 uint_type x_bits = std::bit_cast<uint_type>(x);
86
87 // Extract exponent and mantissa
88 int exponent = static_cast<int>(((x_bits & exponent_mask) >> (std::is_same_v<T, float> ? 23 : 52)) - exponent_bias);
89 uint_type mantissa = x_bits & mantissa_mask;
90
91 // Normalize the mantissa (implicit leading 1 for normalized numbers)
92 if (exponent != -exponent_bias)
93 mantissa |= std::is_same_v<T, float> ? (1u << 23) : (1ull << 52); // Set implicit leading bit
94
95 // Scale the exponent by adding y
96 exponent += static_cast<int>(y);
97
98 // Check for overflow and underflow
99 if (exponent > std::numeric_limits<T>::max_exponent)
100 return std::numeric_limits<T>::infinity();
101
102 if (exponent < std::numeric_limits<T>::min_exponent)
103 return static_cast<T>(0.0); // Underflow to zero
104
105 // Rebuild the floating-point number from the new exponent and mantissa
106 x_bits =
107 (x_bits & (std::is_same_v<T, float> ? 0x80000000u : 0x8000000000000000ull)) | // Preserve the sign bit
108 (static_cast<uint_type>(exponent + exponent_bias) << (std::is_same_v<T, float> ? 23 : 52)) | // Apply new exponent
109 (mantissa & mantissa_mask); // Mask mantissa bits
110
111 return std::bit_cast<T>(x_bits);
112 } else {
113 // Runtime path using fast standard library call
114 return std::scalbn(x, static_cast<int>(y));
115 }
116}
