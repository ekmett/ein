70 EIN(nodiscard,inline,pure)
71constexpr T scalef(T x, T y) noexcept {
72 if consteval {
73 // Constexpr path using bit manipulation
74
75 using uint_type = uint_t<T>; // std::conditional_t<std::is_same_v<T, float>, uint32_t, uint64_t>;
76 constexpr int exponent_bias = std::is_same_v<T, float> ? 127 : 1023;
77 constexpr uint_type exponent_mask = std::is_same_v<T, float> ? uint_type(0x7F800000) : uint_type(0x7FF0000000000000ull);
78 constexpr uint_type mantissa_mask = std::is_same_v<T, float> ? uint_type(0x007FFFFF) : uint_type(0x000FFFFFFFFFFFFFull);
79
80 if (x == 0.0 || isnan(x) || std::isinf(x))
81 return x; // Handle special cases
82
83 // Get the raw bits of the floating-point number
84 uint_type x_bits = std::bit_cast<uint_type>(x);
85
86 // Extract exponent and mantissa
87 int exponent = static_cast<int>(((x_bits & exponent_mask) >> (std::is_same_v<T, float> ? 23 : 52)) - exponent_bias);
88 uint_type mantissa = x_bits & mantissa_mask;
89
90 // Normalize the mantissa (implicit leading 1 for normalized numbers)
91 if (exponent != -exponent_bias)
92 mantissa |= std::is_same_v<T, float> ? (1u << 23) : (1ull << 52); // Set implicit leading bit
93
94 // Scale the exponent by adding y
95 exponent += static_cast<int>(y);
96
97 // Check for overflow and underflow
98 if (exponent > std::numeric_limits<T>::max_exponent)
99 return std::numeric_limits<T>::infinity();
100
101 if (exponent < std::numeric_limits<T>::min_exponent)
102 return static_cast<T>(0.0); // Underflow to zero
103
104 // Rebuild the floating-point number from the new exponent and mantissa
105 x_bits =
106 (x_bits & (std::is_same_v<T, float> ? 0x80000000u : 0x8000000000000000ull)) | // Preserve the sign bit
107 (static_cast<uint_type>(exponent + exponent_bias) << (std::is_same_v<T, float> ? 23 : 52)) | // Apply new exponent
108 (mantissa & mantissa_mask); // Mask mantissa bits
109
110 return std::bit_cast<T>(x_bits);
111 } else {
112 // Runtime path using fast standard library call
113 return std::scalbn(x, static_cast<int>(y));
114 }
115}
