93[[nodiscard]]
94EIN(inline,pure)
95constexpr T scalef(T x, T y) noexcept {
96 if consteval {
97 // Constexpr path using bit manipulation
98
99 using uint_type = uint_t<T>; // std::conditional_t<std::is_same_v<T, float>, uint32_t, uint64_t>;
100 constexpr int exponent_bias = std::is_same_v<T, float> ? 127 : 1023;
101 constexpr uint_type exponent_mask = std::is_same_v<T, float> ? uint_type(0x7F800000) : uint_type(0x7FF0000000000000ull);
102 constexpr uint_type mantissa_mask = std::is_same_v<T, float> ? uint_type(0x007FFFFF) : uint_type(0x000FFFFFFFFFFFFFull);
103
104 if (x == 0.0 || isnan(x) || std::isinf(x))
105 return x; // Handle special cases
106
107 // Get the raw bits of the floating-point number
108 uint_type x_bits = std::bit_cast<uint_type>(x);
109
110 // Extract exponent and mantissa
111 int exponent = static_cast<int>(((x_bits & exponent_mask) >> (std::is_same_v<T, float> ? 23 : 52)) - exponent_bias);
112 uint_type mantissa = x_bits & mantissa_mask;
113
114 // Normalize the mantissa (implicit leading 1 for normalized numbers)
115 if (exponent != -exponent_bias)
116 mantissa |= std::is_same_v<T, float> ? (1u << 23) : (1ull << 52); // Set implicit leading bit
117
118 // Scale the exponent by adding y
119 exponent += static_cast<int>(y);
120
121 // Check for overflow and underflow
122 if (exponent > std::numeric_limits<T>::max_exponent)
123 return std::numeric_limits<T>::infinity();
124
125 if (exponent < std::numeric_limits<T>::min_exponent)
126 return static_cast<T>(0.0); // Underflow to zero
127
128 // Rebuild the floating-point number from the new exponent and mantissa
129 x_bits =
130 (x_bits & (std::is_same_v<T, float> ? 0x80000000u : 0x8000000000000000ull)) | // Preserve the sign bit
131 (static_cast<uint_type>(exponent + exponent_bias) << (std::is_same_v<T, float> ? 23 : 52)) | // Apply new exponent
132 (mantissa & mantissa_mask); // Mask mantissa bits
133
134 return std::bit_cast<T>(x_bits);
135 } else {
136 // Runtime path using fast standard library call
137 return std::scalbn(x, static_cast<int>(y));
138 }
139}
