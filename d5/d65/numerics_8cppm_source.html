60constexpr T scalef(T x, T y) noexcept {
61 if consteval {
62 // Constexpr path using bit manipulation
63
64 using uint_type = uint_t<T>; // std::conditional_t<std::is_same_v<T, float>, uint32_t, uint64_t>;
65 constexpr int exponent_bias = std::is_same_v<T, float> ? 127 : 1023;
66 constexpr uint_type exponent_mask = std::is_same_v<T, float> ? 0x7F800000 : 0x7FF0000000000000ull;
67 constexpr uint_type mantissa_mask = std::is_same_v<T, float> ? 0x007FFFFF : 0x000FFFFFFFFFFFFFull;
68
69 if (x == 0.0 || isnan(x) || std::isinf(x))
70 return x; // Handle special cases
71
72 // Get the raw bits of the floating-point number
73 uint_type x_bits = std::bit_cast<uint_type>(x);
74
75 // Extract exponent and mantissa
76 int exponent = static_cast<int>(((x_bits & exponent_mask) >> (std::is_same_v<T, float> ? 23 : 52)) - exponent_bias);
77 uint_type mantissa = x_bits & mantissa_mask;
78
79 // Normalize the mantissa (implicit leading 1 for normalized numbers)
80 if (exponent != -exponent_bias)
81 mantissa |= std::is_same_v<T, float> ? (1u << 23) : (1ull << 52); // Set implicit leading bit
82
83 // Scale the exponent by adding y
84 exponent += static_cast<int>(y);
85
86 // Check for overflow and underflow
87 if (exponent > std::numeric_limits<T>::max_exponent)
88 return std::numeric_limits<T>::infinity();
89
90 if (exponent < std::numeric_limits<T>::min_exponent)
91 return static_cast<T>(0.0); // Underflow to zero
92
93 // Rebuild the floating-point number from the new exponent and mantissa
94 x_bits =
95 (x_bits & (std::is_same_v<T, float> ? 0x80000000u : 0x8000000000000000ull)) | // Preserve the sign bit
96 (static_cast<uint_type>(exponent + exponent_bias) << (std::is_same_v<T, float> ? 23 : 52)) | // Apply new exponent
97 (mantissa & mantissa_mask); // Mask mantissa bits
98
99 return std::bit_cast<T>(x_bits);
100 } else {
101 // Runtime path using fast standard library call
102 return std::scalbn(x, static_cast<int>(y));
103 }
104}
