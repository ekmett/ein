83[[nodiscard]] EIN(inline,pure)
84constexpr T scalef(T x, T y) noexcept {
85 if consteval {
86 // Constexpr path using bit manipulation
87
88 using uint_type = uint_t<T>; // std::conditional_t<std::is_same_v<T, float>, uint32_t, uint64_t>;
89 constexpr int exponent_bias = std::is_same_v<T, float> ? 127 : 1023;
90 constexpr uint_type exponent_mask = std::is_same_v<T, float> ? uint_type(0x7F800000) : uint_type(0x7FF0000000000000ull);
91 constexpr uint_type mantissa_mask = std::is_same_v<T, float> ? uint_type(0x007FFFFF) : uint_type(0x000FFFFFFFFFFFFFull);
92
93 if (x == 0.0 || isnan(x) || std::isinf(x))
94 return x; // Handle special cases
95
96 // Get the raw bits of the floating-point number
97 uint_type x_bits = std::bit_cast<uint_type>(x);
98
99 // Extract exponent and mantissa
100 int exponent = static_cast<int>(((x_bits & exponent_mask) >> (std::is_same_v<T, float> ? 23 : 52)) - exponent_bias);
101 uint_type mantissa = x_bits & mantissa_mask;
102
103 // Normalize the mantissa (implicit leading 1 for normalized numbers)
104 if (exponent != -exponent_bias)
105 mantissa |= std::is_same_v<T, float> ? (1u << 23) : (1ull << 52); // Set implicit leading bit
106
107 // Scale the exponent by adding y
108 exponent += static_cast<int>(y);
109
110 // Check for overflow and underflow
111 if (exponent > std::numeric_limits<T>::max_exponent)
112 return std::numeric_limits<T>::infinity();
113
114 if (exponent < std::numeric_limits<T>::min_exponent)
115 return static_cast<T>(0.0); // Underflow to zero
116
117 // Rebuild the floating-point number from the new exponent and mantissa
118 x_bits =
119 (x_bits & (std::is_same_v<T, float> ? 0x80000000u : 0x8000000000000000ull)) | // Preserve the sign bit
120 (static_cast<uint_type>(exponent + exponent_bias) << (std::is_same_v<T, float> ? 23 : 52)) | // Apply new exponent
121 (mantissa & mantissa_mask); // Mask mantissa bits
122
123 return std::bit_cast<T>(x_bits);
124 } else {
125 // Runtime path using fast standard library call
126 return std::scalbn(x, static_cast<int>(y));
127 }
128}
